%% ── MODOS ESPECIALES ─────────────────────────────────────
\begin{mdframed}[backgroundcolor=cblue!5,linecolor=cblue!30,linewidth=1pt,roundcorner=6pt]
\textbf{\color{cblue!70!black}Finly Detective}
\hfill\textit{\scriptsize Activa cuando: Ingr $-$ GF $<$ 35\% de ingresos}\\[4pt]
\textcolor{cdark!70}{\small
``¡Oops! Tus gastos fijos est\'an tomando mucho espacio. Pero no te preocupes,
para eso existe Finly Detective: te ayudamos a revisar cada gasto juntas y
encontrar d\'onde puede haber un respiro.''}\\[4pt]
\textbf{\color{cdark!60}\small Objetivo:}
\textcolor{cdark!60}{\small Identificar gastos fijos que se puedan reducir o eliminar.
El an\'alisis detallado se construye en una pr\'oxima versi\'on.}
\end{mdframed}

\vspace{0.3cm}

\begin{mdframed}[backgroundcolor=cred!5,linecolor=cred!30,linewidth=1pt,roundcorner=6pt]
\textbf{\color{cred!70!black}Finly Rompe-deudas}
\hfill\textit{\scriptsize Activa cuando: Ingr $-$ GF $-$ Deudas $\leq$ 0}\\[4pt]
\textcolor{cdark!70}{\small
``Tu situaci\'on actual est\'a enfocada en salir de deudas, y eso es exactamente
lo que vamos a hacer. No hablamos de ahorro todav\'ia --- nos concentramos en
libertad.''}\\[4pt]
\textbf{\color{cdark!60}\small Muestra:} \textcolor{cdark!60}{\small Solo el test de metodolog\'ia de deuda.
Sin cajitas ni bolsitas.}
\end{mdframed}

\vspace{0.4cm}

%% ── CAJITAS ──────────────────────────────────────────────
\begin{mdframed}[backgroundcolor=cbg,linecolor=clpink,linewidth=1pt,roundcorner=6pt]
\textbf{\color{cpink}[4] Cajitas --- Gastos Peri\'odicos}\\[4pt]

\textcolor{cdark!70}{\small ``Hay gastos que no llegan cada mes, pero cuando llegan duelen si no
estamos preparadas. Las cajitas reservan un poquito cada mes para que no te sorprendan y
evitar deudas a futuro.''}

\vspace{6pt}
\begin{minipage}[t]{0.47\textwidth}
\textbf{\color{cdark!70}\small Datos que pide Finly:}
\begin{itemize}[leftmargin=*,itemsep=1pt,topsep=2pt]
  \item[\color{cpink}$\bullet$] {\small Nombre del gasto}
  \item[\color{cpink}$\bullet$] {\small Monto total a pagar}
  \item[\color{cpink}$\bullet$] {\small Fecha de pago}
  \item[\color{cpink}$\bullet$] {\small Finly calcula cuota mensual a reservar}
  \item[\color{cpink}$\bullet$] {\small Notificaci\'on antes del vencimiento}
  \item[\color{cpink}$\bullet$] {\small Al pagar: celebraci\'on + rec\'alculo}
\end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.47\textwidth}
\textbf{\color{cdark!70}\small Ejemplos predefinidos:}
\begin{itemize}[leftmargin=*,itemsep=1pt,topsep=2pt]
  \item[\color{cpink}$\bullet$] {\small SOAT}
  \item[\color{cpink}$\bullet$] {\small Impuesto vehicular / predial}
  \item[\color{cpink}$\bullet$] {\small Seguro del carro / de vida}
  \item[\color{cpink}$\bullet$] {\small Revisi\'on t\'ecnico-mec\'anica}
  \item[\color{cpink}$\bullet$] {\small Matr\'icula escolar / universitaria}
  \item[\color{cpink}$\bullet$] {\small Regalos de navidad y cumplea\~nos}
  \item[\color{cpink}$\bullet$] {\small Membres\'ia anual (gimnasio, etc.)}
  \item[\color{cpink}$\bullet$] {\small Vacaciones anuales}
  \item[\color{cpink}$\bullet$] {\small Mantenimiento del hogar}
\end{itemize}
\end{minipage}
\end{mdframed}

\vspace{0.4cm}

%% ── BOLSITAS ─────────────────────────────────────────────
\begin{mdframed}[backgroundcolor=cgreen!3,linecolor=cgreen!30,linewidth=1pt,roundcorner=6pt]
\textbf{\color{cgreen!60!black}[5] Bolsitas de Ahorro}\\[4pt]
\textcolor{cdark!70}{\small``Ahorrar no es guardar dinero en un caj\'on.
Es darle un prop\'osito a cada peso que decides reservar.''}\\[3pt]
\textcolor{cdark!60}{\scriptsize \textit{Nota: Finly calcula y recomienda. La usuaria crea las cuentas/bolsillos en su banco real
--- los montos m\'inimos los define cada banco.}}

\vspace{6pt}
\textbf{\color{cdark!70}\small Modo A --- Sin bolsitas (ahorro general):}\\[2pt]
\textcolor{cdark!60}{\small La usuaria define un monto \'unico de ahorro mensual. Simple y sin categor\'ias.}

\vspace{6pt}
\textbf{\color{cdark!70}\small Modo B --- Con bolsitas (ahorro por prop\'osito):}\\[4pt]

\begin{minipage}[t]{0.47\textwidth}
\textbf{\color{cdark!70}\small Mis fondos \textnormal{\scriptsize(bolsitas permanentes)}}\\[2pt]
\textcolor{cdark!60}{\scriptsize Siempre est\'an contigo. Se llenan mes a mes sin fecha de cierre.}\\[4pt]
\textbf{\color{cdark!60}\scriptsize Predefinidas:}
\begin{itemize}[leftmargin=*,itemsep=1pt,topsep=2pt]
  \item[\color{cgreen!60!black}$\bullet$] {\small Skincare / belleza}
  \item[\color{cgreen!60!black}$\bullet$] {\small Regalos}
  \item[\color{cgreen!60!black}$\bullet$] {\small Emergencias del hogar}
  \item[\color{cgreen!60!black}$\bullet$] {\small Emergencias del carro}
  \item[\color{cgreen!60!black}$\bullet$] {\small Salud / medicamentos}
  \item[\color{cgreen!60!black}$\bullet$] {\small + Agregar personalizada}
\end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.47\textwidth}
\textbf{\color{cdark!70}\small Mis metas \textnormal{\scriptsize(bolsitas con fecha)}}\\[2pt]
\textcolor{cdark!60}{\scriptsize Tienen costo total y fecha. Finly calcula el ahorro mensual necesario.}\\[4pt]
\textbf{\color{cdark!60}\scriptsize Ejemplos:}
\begin{itemize}[leftmargin=*,itemsep=1pt,topsep=2pt]
  \item[\color{cgreen!60!black}$\bullet$] {\small Viaje (Disney, Cartagena...)}
  \item[\color{cgreen!60!black}$\bullet$] {\small Celular nuevo}
  \item[\color{cgreen!60!black}$\bullet$] {\small Computador}
  \item[\color{cgreen!60!black}$\bullet$] {\small Bolso / ropa especial}
  \item[\color{cgreen!60!black}$\bullet$] {\small Keratina / tratamiento}
  \item[\color{cgreen!60!black}$\bullet$] {\small Carro / moto}
\end{itemize}
\end{minipage}
